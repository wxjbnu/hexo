//递归树
fnTree (data, pid) {
    var result = [], temp;
    for (var i = 0; i < data.length; i++) {
        if (data[i].parentId == pid) {
            var obj = {"label": data[i].title,"id": data[i].id};
            temp = this.fnTree(data, data[i].id);
            if (temp.length > 0) {
                obj.children = temp;
            }
            result.push(obj);
        }
    }
    return result;
},
